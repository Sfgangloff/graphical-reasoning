\section*{Tag vocabulary}

Each problem below carries one or more of the following tags. They cover
limits/convergence, continuity, differentiability, integrability,
inequalities involving derivatives or integrals, ODEs on the real line, and
asymptotic behavior of real-valued functions. Problems whose substance is
combinatorics, number theory, algebra, geometry, probability, complex
analysis, or formal power series with combinatorial flavor were excluded.

\begin{description}[leftmargin=2.5cm,style=nextline]
\item[\texttt{limit-sequence}] Limits of sequences of real numbers.
\item[\texttt{limit-series}] Convergence/sum of series of real numbers.
\item[\texttt{continuity}] Continuity properties of real functions.
\item[\texttt{differentiability}] Differentiability properties; uses of derivatives.
\item[\texttt{integrability}] Integrability questions / structural results about integrals.
\item[\texttt{integral-evaluation}] Compute a definite or improper real integral.
\item[\texttt{integral-inequality}] Inequality involving one or more integrals.
\item[\texttt{derivative-inequality}] Inequality involving one or more derivatives.
\item[\texttt{ode}] Ordinary differential equation on the real line.
\item[\texttt{asymptotic}] Asymptotic behavior of real-valued sequences/functions.
\item[\texttt{convergence}] Convergence questions not fitting limit-sequence/series exactly.
\item[\texttt{iteration}] Iterated maps / recurrences whose question is the limit behavior.
\item[\texttt{functional-real}] Functional equation/property of a real-valued function.
\end{description}

\section*{1985}

\paragraph{1985 A--3 \textnormal{[\texttt{limit-sequence, iteration}]}}
Let $d$ be a real number. For each integer $m \geq 0$, define a
sequence $\{a_m(j)\}$, $j=0,1,2,\dots$ by the condition
\begin{align*}
a_m(0) &= d/2^m, \\
a_m(j+1) &= (a_m(j))^2 + 2a_m(j), \qquad j \geq 0.
\end{align*}
Evaluate $\lim_{n \to \infty} a_n(n)$.

\paragraph{1985 B--5 \textnormal{[\texttt{integral-evaluation}]}}
Evaluate $\int_0^\infty t^{-1/2}e^{-1985(t+t^{-1})}\,dt$. You may
assume that $\int_{-\infty}^\infty e^{-x^2}\,dx = \sqrt{\pi}$.

\section*{1986}

\paragraph{1986 A--1 \textnormal{[\texttt{differentiability}]}}
Find, with explanation, the maximum value of $f(x)=x^3-3x$ on the
set of all real numbers $x$ satisfying $x^4+36\leq 13x^2$.

\paragraph{1986 A--3 \textnormal{[\texttt{limit-series}]}}
Evaluate $\sum_{n=0}^\infty \mathrm{Arccot}(n^2+n+1)$, where
$\mathrm{Arccot}\,t$ for $t \geq 0$ denotes the number $\theta$ in the
interval $0 < \theta \leq \pi/2$ with $\cot \theta = t$.

\section*{1987}

\paragraph{1987 A--3 \textnormal{[\texttt{ode, differentiability, derivative-inequality}]}}
For all real $x$, the real-valued function $y=f(x)$ satisfies
\[
y''-2y'+y=2e^x.
\]
\begin{enumerate}
\item[(a)] If $f(x)>0$ for all real $x$, must $f'(x) > 0$ for all real $x$? Explain.
\item[(b)] If $f'(x)>0$ for all real $x$, must $f(x) > 0$ for all real $x$? Explain.
\end{enumerate}

\paragraph{1987 A--6 \textnormal{[\texttt{limit-series, convergence}]}}
For each positive integer $n$, let $a(n)$ be the number of zeroes in
the base 3 representation of $n$. For which positive real numbers $x$
does the series
\[
\sum_{n=1}^\infty \frac{x^{a(n)}}{n^3}
\]
converge?

\paragraph{1987 B--1 \textnormal{[\texttt{integral-evaluation}]}}
Evaluate
\[
\int_2^4 \frac{\sqrt{\ln(9-x)}\,dx}{\sqrt{\ln(9-x)}+\sqrt{\ln(x+3)}}.
\]

\paragraph{1987 B--4 \textnormal{[\texttt{limit-sequence, iteration}]}}
Let $(x_1,y_1) = (0.8, 0.6)$ and let $x_{n+1} = x_n \cos y_n - y_n
\sin y_n$ and $y_{n+1}= x_n \sin y_n + y_n \cos y_n$ for
$n=1,2,3,\dots$. For each of $\lim_{n\to \infty} x_n$ and $\lim_{n \to
\infty} y_n$, prove that the limit exists and find it or prove that
the limit does not exist.

\section*{1988}

\paragraph{1988 A--2 \textnormal{[\texttt{differentiability, functional-real}]}}
A not uncommon calculus mistake is to believe that the product rule
for derivatives says that $(fg)' = f'g'$. If $f(x)=e^{x^2}$,
determine, with proof, whether there exists an open interval $(a,b)$
and a nonzero function $g$ defined on $(a,b)$ such that this wrong
product rule is true for $x$ in $(a,b)$.

\paragraph{1988 A--3 \textnormal{[\texttt{limit-series, convergence}]}}
Determine, with proof, the set of real numbers $x$ for which
\[
\sum_{n=1}^\infty \left( \frac{1}{n} \csc \frac{1}{n} - 1 \right)^x
\]
converges.

\paragraph{1988 B--4 \textnormal{[\texttt{limit-series, convergence}]}}
Prove that if $\sum_{n=1}^\infty a_n$ is a convergent series of
positive real numbers, then so is $\sum_{n=1}^\infty (a_n)^{n/(n+1)}$.

\section*{1989}

\paragraph{1989 A--2 \textnormal{[\texttt{integral-evaluation}]}}
Evaluate
$\displaystyle{\int_0^a\int_0^b e^{{\rm max}\{b^2x^2, a^2y^2\}}\,dy\,dx}$
where $a$ and $b$ are positive.

\paragraph{1989 B--3 \textnormal{[\texttt{ode, integral-evaluation, asymptotic}]}}
Let $f$ be a function on $[0,\infty)$, differentiable and satisfying
\[
f'(x)=-3f(x)+6f(2x)
\]
for $x>0$. Assume that $|f(x)|\le e^{-\sqrt{x}}$ for $x\ge 0$ (so that
$f(x)$ tends rapidly to $0$ as $x$ increases).
For $n$ a non-negative integer, define
\[
\mu_n=\int_0^\infty x^n f(x)\,dx
\]
(sometimes called the $n$th moment of $f$).
\begin{enumerate}
\item[a)] Express $\mu_n$ in terms of $\mu_0$.
\item[b)] Prove that the sequence $\{\mu_n \frac{3^n}{n!}\}$ always converges,
and that the limit is $0$ only if $\mu_0=0$.
\end{enumerate}

\paragraph{1990 B--1 \textnormal{[\texttt{differentiability, integral-evaluation, ode}]}}

on the real line such that for all $x$,

\]

\section*{1991}

\paragraph{1991 A--5 \textnormal{[\texttt{integral-evaluation}]}}
Find the maximum value of
\[
\int_0^y \sqrt{x^4 + (y-y^2)^2}\,dx
\]
for $0 \leq y \leq 1$.

\paragraph{1991 B--2 \textnormal{[\texttt{differentiability, functional-real}]}}
Suppose $f$ and $g$ are non-constant, differentiable, real-valued
functions defined on $(-\infty, \infty)$. Furthermore, suppose that for
each pair of real numbers $x$ and $y$,
\begin{align*}
f(x+y) &= f(x)f(y) - g(x)g(y), \\
g(x+y) &= f(x)g(y) + g(x)f(y).
\end{align*}
If $f'(0) = 0$, prove that $(f(x))^2 + (g(x))^2 = 1$ for all $x$.

\paragraph{1991 B--6 \textnormal{[\texttt{integral-inequality}]}}
Let $a$ and $b$ be positive numbers. Find the largest number $c$, in
terms of $a$ and $b$, such that
\[
a^x b^{1-x} \leq a \frac{\sinh ux}{\sinh u} + b \frac{\sinh u(1-x)}{\sinh u}
\]
for all $u$ with $0 < |u| \leq c$ and for all $x$, $0 < x < 1$. (Note:
$\sinh u = (e^u - e^{-u})/2$.)

\section*{1992}

\paragraph{1992 A--4 \textnormal{[\texttt{differentiability, limit-sequence}]}}
Let $f$ be an infinitely differentiable real-valued function
defined on the real numbers. If
\[
f\left( \frac{1}{n} \right) = \frac{n^2}{n^2 + 1}, \qquad n = 1, 2, 3, \dots,
\]
compute the values of the derivatives $f^{(k)}(0), k = 1, 2, 3, \dots$.

\paragraph{1992 B--3 \textnormal{[\texttt{iteration, convergence}]}}
For any pair $(x, y)$ of real numbers, a sequence
$(a_n(x,y))_{n\geq 0}$ is defined as follows:
\begin{align*}
a_0(x, y) &= x, \\
a_{n+1}(x, y) &= \frac{(a_n(x, y))^2 + y^2}{2}, \qquad \mbox{for $n \geq 0$}.
\end{align*}
Find the area of the region
\[
\{ (x, y) | (a_n(x, y))_{n \geq 0}\ \mbox{converges}\}.
\]

in the first quadrant as in the figure. Find $c$ so that the areas of the
two shaded regions are equal. [Figure not included. The first region is

\paragraph{1993 A--2 \textnormal{[\texttt{iteration}]}}
Let $(x_n)_{n \geq 0}$ be a sequence of nonzero real numbers
such that $x_n^2 - x_{n-1}x_{n+1} = 1$ for $n=1,2,3,\dots$. Prove there
exists a real number $a$ such that $x_{n+1} = ax_n - x_{n-1}$ for all $n
\geq 1$.

\paragraph{1993 A--5 \textnormal{[\texttt{integral-evaluation}]}}
Show that
\begin{gather*}
\int_{-100}^{-10} \left( \frac{x^2 - x}{x^3 - 3x + 1} \right)^2\,dx + \\
\int_{\frac{1}{101}}^{\frac{1}{11}} \left( \frac{x^2 - x}{x^3 - 3x + 1} \right)^2\,dx + \\
\int_{\frac{101}{100}}^{\frac{11}{10}} \left( \frac{x^2 - x}{x^3 - 3x + 1} \right)^2\,dx
\end{gather*}
is a rational number.

\paragraph{1993 B--4 \textnormal{[\texttt{continuity, integral-evaluation, functional-real}]}}
The function $K(x,y)$ is positive and continuous for $0 \leq x \leq 1, 0
\leq y \leq 1$, and the functions $f(x)$ and $g(x)$ are positive and
continuous for $0 \leq x \leq 1$. Suppose that for all $x$, $0 \leq x \leq 1$,
\[
\int_0^1 f(y)K(x,y)\,dy = g(x)
\]
and
\[
\int_0^1 g(y)K(x,y)\,dy = f(x).
\]
Show that $f(x) = g(x)$ for $0 \leq x \leq 1$.

\section*{1994}

\paragraph{1994 A--1 \textnormal{[\texttt{limit-series, convergence}]}}
Suppose that a sequence $a_1, a_2, a_3, \dots$ satisfies
$0 < a_n \leq a_{2n} + a_{2n+1}$ for all $n \geq 1$. Prove that the series
$\sum_{n=1}^{\infty} a_n$ diverges.

\paragraph{1994 B--3 \textnormal{[\texttt{differentiability, derivative-inequality, asymptotic}]}}
Find the set of all real numbers $k$ with the following property: For any
positive, differentiable function $f$ that satisfies $f'(x) > f(x)$
for all $x$, there is some number $N$ such that
$f(x) > e^{kx}$ for all $x > N$.

\section*{1995}

\paragraph{1995 A--2 \textnormal{[\texttt{integral-evaluation, convergence}]}}
For what pairs $(a,b)$ of positive real numbers does the
improper integral
\[
\int_{b}^{\infty} \left( \sqrt{\sqrt{x+a}-\sqrt{x}} -
\sqrt{\sqrt{x}-\sqrt{x-b}} \right)\,dx
\]

\paragraph{1995 A--5 \textnormal{[\texttt{ode, asymptotic}]}}
Let $x_{1},x_{2},\dots,x_{n}$ be differentiable
(real-valued) functions of a single variable $t$ which satisfy
\begin{align*}
\frac{dx_{1}}{dt} &= a_{11}x_{1} + a_{12}x_{2} + \cdots + a_{1n}x_{n} \\
\frac{dx_{2}}{dt} &= a_{21}x_{1} + a_{22}x_{2} + \cdots + a_{2n}x_{n} \\
&\vdots \\
\frac{dx_{n}}{dt} &= a_{n1}x_{1} + a_{n2}x_{2} + \cdots + a_{nn}x_{n}
\end{align*}
for some constants $a_{ij}>0$. Suppose that for all $i$, $x_{i}(t)
\to 0$ as $t \to \infty$. Are the functions $x_{1},x_{2},\dots,x_{n}$
necessarily linearly dependent?

\paragraph{1996 B--2 \textnormal{[\texttt{asymptotic, integral-inequality}]}}

\cdots (2n-1) < \left( \frac{2n+1}{e} \right)^{\frac{2n+1}{2}}.
\]

4}-\frac{x^7}{2\cdot 4\cdot 6}+\cdots\right) \\
\left(1+\frac{x^2}{2^2}+

\section*{1998}

\paragraph{1998 A--2 \textnormal{[\texttt{integral-evaluation}]}}
Let $s$ be any arc of the unit circle lying entirely in the first
quadrant. Let $A$ be the area of the region lying below $s$ and
above the $x$-axis and let $B$ be the area of the region lying to the
right of the $y$-axis and to the left of $s$. Prove that $A+B$ depends
only on the arc length, and not on the position, of $s$.

\paragraph{1998 A--3 \textnormal{[\texttt{differentiability, derivative-inequality}]}}
Let $f$ be a real function on the real line with continuous third
derivative. Prove that there exists a point $a$ such that
\[f(a)\cdot f'(a) \cdot f''(a) \cdot f'''(a)\geq 0 .\]

\section*{1999}

\paragraph{1999 A--5 \textnormal{[\texttt{integral-inequality}]}}
Prove that there is a constant $C$ such that, if $p(x)$ is a polynomial
of degree 1999, then
\[|p(0)|\leq C \int_{-1}^1 |p(x)|\,dx.\]

\paragraph{1999 B--1 \textnormal{[\texttt{limit-sequence}]}}
Right triangle $ABC$ has right angle at $C$ and $\angle BAC =\theta$;
the point $D$ is chosen on $AB$ so that $|AC|=|AD|=1$; the point $E$
is chosen on $BC$ so that $\angle CDE = \theta$. The perpendicular
to $BC$ at $E$ meets $AB$ at $F$. Evaluate $\lim_{\theta\rightarrow 0}
|EF|$.

\paragraph{1999 B--3 \textnormal{[\texttt{limit-series}]}}
Let $A=\{(x,y):0\leq x,y<1\}$. For $(x,y)\in A$, let
\[S(x,y) = \sum_{\frac{1}{2}\leq \frac{m}{n}\leq 2} x^m y^n,\]
where the sum ranges over all pairs $(m,n)$ of positive integers
satisfying the indicated inequalities. Evaluate
\[\lim_{(x,y)\rightarrow (1,1), (x,y)\in A} (1-xy^2)(1-x^2y)S(x,y).\]

\paragraph{1999 B--4 \textnormal{[\texttt{differentiability, derivative-inequality}]}}
Let $f$ be a real function with a continuous third derivative such that $f(x),
f'(x), f''(x), f'''(x)$ are positive for all $x$. Suppose that
$f'''(x)\leq f(x)$ for all $x$. Show that $f'(x)<2f(x)$ for all $x$.

\section*{2000}

\paragraph{2000 A--1 \textnormal{[\texttt{convergence, limit-series}]}}
Let $A$ be a positive real number. What are the possible values of
$\sum_{j=0}^\infty x_j^2$, given that $x_0,x_1,\ldots$ are positive
numbers
for which $\sum_{j=0}^\infty x_j=A$?

\paragraph{2000 A--4 \textnormal{[\texttt{integral-evaluation, convergence}]}}
Show that the improper integral
\[ \lim_{B\to\infty}\int_{0}^B \sin(x) \sin(x^2)\,dx\]
converges.

\paragraph{2000 B--3 \textnormal{[\texttt{differentiability, asymptotic}]}}
Let $f(t)=\sum_{j=1}^N a_j \sin(2\pi jt)$, where each $a_j$ is real
and
$a_N$ is not equal to 0. Let $N_k$ denote the number of zeroes (including
multiplicities) of $\frac{d^k f}{dt^k}$.
Prove that
\[N_0\leq N_1\leq N_2\leq \cdots \mbox{ and } \lim_{k\to\infty} N_k = 2N.\]
[Editorial clarification: only zeroes in $[0, 1)$ should be counted.]

\paragraph{2000 B--4 \textnormal{[\texttt{continuity, functional-real}]}}
Let $f(x)$ be a continuous function such that $f(2x^2-1)=2xf(x)$ for
all $x$. Show that $f(x)=0$ for $-1\leq x\leq 1$.

\section*{2002}

\section*{2004}

\paragraph{2004 A--1 \textnormal{[\texttt{continuity}]}}
Basketball star Shanille O'Keal's team statistician
keeps track of the number, $S(N)$, of successful free throws she has made
in her first $N$ attempts of the season.
Early in the season, $S(N)$  was less than 80\% of $N$,
but by the end of the season, $S(N)$ was more than 80\% of $N$.
Was there necessarily a moment in between when $S(N)$ was exactly 80\% of $N$?

\paragraph{2004 A--6 \textnormal{[\texttt{integral-inequality}]}}
Suppose that $f(x,y)$ is a continuous real-valued function on the unit
square $0 \le x \le 1, 0 \le y \le 1$. Show that
\begin{align*}
& \int_0^1 \left( \int_0^1  f(x,y) dx \right)^2 dy +
 \int_0^1 \left( \int_0^1  f(x,y) dy \right)^2 dx \\
&\leq
\left( \int_0^1 \int_0^1  f(x,y) dx\, dy \right)^2 +
\int_0^1 \int_0^1 \left[ f(x,y) \right]^2 dx\,dy.
\end{align*}

\paragraph{2004 B--3 \textnormal{[\texttt{continuity, integral-evaluation}]}}
