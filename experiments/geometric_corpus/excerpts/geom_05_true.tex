\begin{problem}[Geometric claim 05 --- variant: TRUE]
Define the sequence
\[
  a_{n} \;=\; \frac{1}{n} \;+\; \frac{1}{200}\,
  \exp\!\Bigl(-\tfrac{(n - 47)^{2}}{2}\Bigr),
  \qquad n = 1, 2, 3, \dots
\]

\textbf{Claim.} The sequence $(a_{n})_{n \geq 1}$ is \emph{not}
monotonically decreasing: there exists $n \geq 1$ for which
$a_{n+1} \geq a_{n}$.
\end{problem}
