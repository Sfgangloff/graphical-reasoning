\begin{problem}[Geometric claim 01 --- variant: TRUE]
Consider the functions
\[
  f(x) = x^{2} + \tfrac{1}{2}
  \qquad \text{and} \qquad
  g(x) = x^{2} + \tfrac{1}{2} - 2\,(x - 0.43)^{2}
\]
on the interval $[0, 1]$.

\textbf{Claim.} On $[0, 1]$ we have $f(x) \geq g(x)$ everywhere, with
equality at \emph{exactly one} point in the open interval $(0, 1)$.
\end{problem}
