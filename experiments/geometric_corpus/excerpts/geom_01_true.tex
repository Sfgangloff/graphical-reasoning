\begin{problem}[Geometric claim 01 — variant: TRUE]
Consider the functions
\[
  f(x) = \sin(\pi x) \qquad \text{and} \qquad g(x) = x^2 (2 - x)
\]
on the interval $[0, 1]$.

\textbf{Claim.} On the open interval $(0, 1)$ the equation
$f(x) = g(x)$ has \emph{exactly one} solution, and at that solution
the curves \emph{cross transversally} (i.e.\ $f - g$ changes sign).
\end{problem>
