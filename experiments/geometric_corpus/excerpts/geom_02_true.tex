\begin{problem}[Geometric claim 02 --- variant: TRUE]
Define the planar parametric curve
\[
  \gamma : [0, 2\pi] \to \mathbb{R}^{2},
  \qquad
  \gamma(t) = \bigl(\cos t,\; \sin 2t\bigr).
\]

\textbf{Claim.} The curve $\gamma$ has a \emph{self-intersection}:
there exist $t_1, t_2 \in [0, 2\pi)$ with $t_1 \neq t_2$ and
$\gamma(t_1) = \gamma(t_2)$.
\end{problem}
