\begin{problem}[Geometric claim 10 --- variant: TRUE]
Define
\[
  f(x) \;=\; (x^{2} - 1)\,(x^{2} - 4) \;+\; 3\sin(50 x),
  \qquad x \in [-2.5,\, 2.5].
\]

\textbf{Claim.} The function $f$ has \emph{strictly more than four}
distinct real zeros in the interval $[-2.5,\, 2.5]$ --- in fact,
several dozen.
\end{problem}
