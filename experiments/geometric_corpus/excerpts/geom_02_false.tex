\begin{problem}[Geometric claim 02 --- variant: FALSE]
Define the planar parametric curve
\[
  \gamma : [0, 2\pi] \to \mathbb{R}^{2},
  \qquad
  \gamma(t) = \bigl(\cos t,\; \sin 2t\bigr).
\]

\textbf{Claim.} The curve $\gamma$ is \emph{simple}: for every pair
$t_1, t_2 \in [0, 2\pi)$ with $t_1 \neq t_2$, $\gamma(t_1) \neq
\gamma(t_2)$.
\end{problem}
