\begin{problem}[Geometric claim 05 --- variant: FALSE]
Define the sequence
\[
  a_{n} \;=\; \frac{1}{n} \;+\; \frac{1}{200}\,
  \exp\!\Bigl(-\tfrac{(n - 47)^{2}}{2}\Bigr),
  \qquad n = 1, 2, 3, \dots
\]

\textbf{Claim.} The sequence $(a_{n})_{n \geq 1}$ is \emph{strictly
monotonically decreasing}: $a_{n+1} < a_{n}$ for every $n \geq 1$.
\end{problem}
