\begin{problem}[Geometric claim 09 --- variant: TRUE]
Define $f : \mathbb{R} \to \mathbb{R}$ by
\[
  f(x) =
  \begin{cases}
    x \cdot \sin\!\bigl(\pi / x\bigr), & x \neq 0, \\
    0, & x = 0.
  \end{cases}
\]

\textbf{Claim.} The function $f$ is \emph{not differentiable} at
$x = 0$.
\end{problem}
