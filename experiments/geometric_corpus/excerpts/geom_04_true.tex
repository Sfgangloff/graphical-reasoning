\begin{problem}[Geometric claim 04 --- variant: TRUE]
Define the region
\[
  R \;=\; \bigl\{\,(x, y) \in \mathbb{R}^{2} \;:\;
    x^{2} + y^{2} \leq 1
    \text{ and }
    (x - 0.27)^{2} + (y - 0.13)^{2} \geq (0.05)^{2}\,\bigr\}.
\]

\textbf{Claim.} $R$ is \emph{not} simply connected: there exists a
continuous loop in $R$ that cannot be continuously contracted to a
point inside $R$.
\end{problem}
