\begin{problem}[Geometric claim 06 --- variant: TRUE]
Define
\[
  h(x) \;=\; \bigl(e^{x} - 1 - x - \tfrac{x^{2}}{2}\bigr)
            \;-\; \tfrac{2}{100}\,\exp\!\bigl(-10000\,(x - 0.43)^{2}\bigr).
\]

\textbf{Claim.} The inequality $h(x) \geq 0$ \emph{fails} on $[0, 1]$:
there exists $x_{0} \in [0, 1]$ with $h(x_{0}) < 0$.
\end{problem}
